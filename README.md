#include <stdio.h>

int main() {
    float x;
    puts("Верно ли высказывание, что x лежит вне отрезков [2, 5] и [-1, 1]");
    printf("Введите X: ");
    scanf("%f", &x);

    if ((x >= 2 && x <= 5) or (x >= -1 && x <= 1))
        printf("Число лежит в отрезке => высказывание неверно.");
    else
        printf("Число не лежит в отрезке => высказывание верное");

    return 0;
}
